def sherlockAndAnagrams(s):
    lista_literek = list(s)
    wszyskie = []
    for i in range(len(lista_literek)):
        for j in range(i, len(lista_literek)):
            wszyskie.append(lista_literek[i:j+1])
    for i in wszyskie:
        i.sort()
    wynik = 0
    for i in range(len(wszyskie)):
        for j in range(i+1, len(wszyskie)):
            if wszyskie[i] == wszyskie[j]: 
                wynik += 1
    
    return wynik
    

if __name__ == '__main__':
    fptr = open(os.environ['OUTPUT_PATH'], 'w')

    q = int(input().strip())

    for q_itr in range(q):
        s = input()

        result = sherlockAndAnagrams(s)

        fptr.write(str(result) + '\n')

    fptr.close()
